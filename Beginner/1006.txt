static void Main(string[] args) 
{ 
	double grade1 = Double.Parse(Console.ReadLine(), System.Globalization.CultureInfo.InvariantCulture);
	double grade2 = Double.Parse(Console.ReadLine(), System.Globalization.CultureInfo.InvariantCulture);
	double grade3 = Double.Parse(Console.ReadLine(), System.Globalization.CultureInfo.InvariantCulture);
	double average = ((grade1 * 2) + (grade2 * 3) + (grade3 * 5)) / 10;

	Console.WriteLine("MEDIA = " + Math.Round(average, 1).ToString("F1"));
}
